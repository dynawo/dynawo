30 DataInterfaceIIDM::configureQuadripoleCriteria(const std::shared_ptr<criteria::CriteriaCollection>& criteria) {
31  for (const auto& quadripoleCriteria : criteria->getQuadripoleCriteria()) {
32  if (!QuadripoleCriteria<T>::criteriaEligibleForQuadripole(quadripoleCriteria->getParams())) continue;
33  std::unique_ptr<QuadripoleCriteria<T>> dynCriteria = DYN::make_unique<QuadripoleCriteria<T>>(quadripoleCriteria->getParams());
34  const auto& componentIds = quadripoleCriteria->getComponentIds();
35  if (!componentIds.empty()) {
36  for (const auto& componentId : componentIds) {
37  std::shared_ptr<T> quadripole;
38  std::unordered_map<std::string, std::shared_ptr<ComponentInterface> >::const_iterator quadripoleItfIter = components_.find(componentId->getId());
39  if (quadripoleItfIter != components_.end()) {
40  const std::shared_ptr<ComponentInterface>& cmp = quadripoleItfIter->second;
41  if (cmp->getType() != ComponentInterface::LINE && cmp->getType() != ComponentInterface::TWO_WTFO) {
42  Trace::warn() << DYNLog(WrongComponentType, componentId->getId(), "quadripole (line or transformer)") << Trace::endline;
43  continue;
44  }
45  if (cmp->getType() != T::derivedType()) {
46  continue;
47  }
48  quadripole = boost::dynamic_pointer_cast<T>(cmp);
49  assert(quadripole);
50  } else {
51  Trace::warn() << DYNLog(ComponentNotFound, componentId->getId()) << Trace::endline;
52  continue;
53  }
54  dynCriteria->addQuadripole(quadripole);
55  }
56  } else {
57  continue;
58  }
59  if (!dynCriteria->empty()) {
60  criteria_.push_back(std::move(dynCriteria));
61  }
62  }
63 }
