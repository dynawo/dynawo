442  inline static double ir_dUr(const double ur, const double ui, const double U2,
443  const double p, const double q, const double PdUr, const double QdUr) {
444  return ((PdUr * ur + p) + (QdUr * ui) - 2. * ur * (p * ur + q * ui) / U2) / U2;
445  }
446 
458  inline static double ii_dUr(const double ur, const double ui, const double U2,
459  const double p, const double q, const double PdUr, const double QdUr) {
460  return ((PdUr * ui) - (QdUr * ur + q) - 2. * ur * (p * ui - q * ur) / U2) / U2;
461  }
462 
474  inline static double ir_dUi(const double ur, const double ui, const double U2,
475  const double p, const double q, const double PdUi, const double QdUi) {
476  return ((PdUi * ur) + (QdUi * ui + q) - 2. * ui * (p * ur + q * ui) / U2) / U2;
477  }
