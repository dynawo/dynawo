41 template<typename T> bool
42 QuadripoleCriteria<T>::checkCriteria(double t, bool finalStep, const boost::shared_ptr<timeline::Timeline>& timeline) {
43  failingCriteria_.clear();
44  assert(params_->getType() != criteria::CriteriaParams::SUM);
45  if (!finalStep && params_->getScope() == criteria::CriteriaParams::FINAL)
46  return true;
47  std::multimap<double, std::unique_ptr<FailingCriteria> > distanceToQuadripoleFailingCriteriaMap;
48  for (const auto& quadripole : quadripoles_) {
49  double i1 = quadripole->getStateVarI1();
50  double i2 = quadripole->getStateVarI2();
51  if (params_->hasI1Max() && i1 > params_->getI1Max()) {
52  std::unique_ptr<FailingCriteria> quadripoleFailingCriteria = DYN::make_unique<QuadripoleFailingCriteria>(Side::ONE,
53  quadripole->getID(),
54  i1,
55  params_->getI1Max(),
56  params_->getId());
57  distanceToQuadripoleFailingCriteriaMap.insert(std::make_pair(quadripoleFailingCriteria->getDistance(), std::move(quadripoleFailingCriteria)));
58  }
59  if (params_->hasI2Max() && i2 > params_->getI2Max()) {
60  std::unique_ptr<FailingCriteria> quadripoleFailingCriteria = DYN::make_unique<QuadripoleFailingCriteria>(Side::TWO,
61  quadripole->getID(),
62  i2,
63  params_->getI2Max(),
64  params_->getId());
65  distanceToQuadripoleFailingCriteriaMap.insert(std::make_pair(quadripoleFailingCriteria->getDistance(), std::move(quadripoleFailingCriteria)));
66  }
67  }
68  if (!distanceToQuadripoleFailingCriteriaMap.empty()) {
69  printAllFailingCriteriaIntoLog(distanceToQuadripoleFailingCriteriaMap, timeline, t);
70  }
71  return distanceToQuadripoleFailingCriteriaMap.empty();
72 }

74 template<typename T> bool
75 QuadripoleCriteria<T>::criteriaEligibleForQuadripole(const std::shared_ptr<criteria::CriteriaParams>& params) {
76  if (params->getType() == criteria::CriteriaParams::SUM) {
77  Trace::warn() << DYNLog(SumQuadripoleCriteriaIgnored) << Trace::endline;
78  return false;
79  }
80  if (params->hasVoltageLevels())
81  return false;
82  if (!params->hasI1Max() && !params->hasI2Max())
83  return false;
84  if (params->hasPMax())
85  Trace::warn() << DYNLog(PowerQuadripoleCriteriaIgnored) << Trace::endline;
86  if (params->hasPMin())
87  Trace::warn() << DYNLog(PowerQuadripoleCriteriaIgnored) << Trace::endline;
88  return true;
89 }

121 template<typename T> void
122 QuadripoleCriteria<T>::QuadripoleFailingCriteria::printOneFailingCriteriaIntoTimeline(const boost::shared_ptr<timeline::Timeline>& timeline,
123  std::vector<std::pair<double, std::string> >& failingCriteria,
124  double currentTime) const {
125  const std::string name = "Simulation";
126  switch (side_) {
127  case DYN::QuadripoleCriteria<T>::Side::ONE: {
128  if (timeline != nullptr) {
129  MessageTimeline messageTimeline = DYNTimeline(QuadripoleAboveCurrent, nodeId_, "1", i_, iBound_, criteriaId_);
130  timeline->addEvent(currentTime, name, messageTimeline.str(), messageTimeline.priority(), messageTimeline.getKey());
131  }
132  Message failingCriteriaMessage = DYNLog(QuadripoleAboveCurrent, nodeId_, "1", i_, iBound_, criteriaId_);
133  failingCriteria.push_back(std::make_pair(currentTime, failingCriteriaMessage.str()));
134  }
135  break;
136  case DYN::QuadripoleCriteria<T>::Side::TWO: {
137  if (timeline != nullptr) {
138  MessageTimeline messageTimeline = DYNTimeline(QuadripoleAboveCurrent, nodeId_, "2", i_, iBound_, criteriaId_);
139  timeline->addEvent(currentTime, name, messageTimeline.str(), messageTimeline.priority(), messageTimeline.getKey());
140  }
141  Message failingCriteriaMessage = DYNLog(QuadripoleAboveCurrent, nodeId_, "2", i_, iBound_, criteriaId_);
142  failingCriteria.push_back(std::make_pair(currentTime, failingCriteriaMessage.str()));
143  }
144  break;
145  }
146 }
