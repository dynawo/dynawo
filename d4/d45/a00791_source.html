81  template<class T>
82  static ExtensionDefinition<T> getExtension(const std::string& libPath) {
83  if (libPath.empty()) {
84  // This corresponds to default case if environment variable for external IIDM extensions paths is not set
85  return buildDefaultExtensionDefinition<T>();
86  }
87  const auto& name = IIDMExtensionTrait<T>::name;
88  std::shared_ptr<boost::dll::shared_library> extensionLibrary;
89  std::unique_lock<std::mutex> lock(librariesMutex_);
90  if (getLibraries().count(libPath) > 0) {
91  extensionLibrary = getLibraries().at(libPath);
92  } else {
93  try {
94  extensionLibrary = std::make_shared<boost::dll::shared_library>(libPath);
95  } catch (const std::exception& e) {
96  if (getLibrariesLoadingIssues().count(libPath) == 0) {
97  // put the log only once
98  Trace::warn() << DYNLog(IIDMExtensionLibraryNotLoaded, libPath, name, e.what()) << Trace::endline;
99  getLibrariesLoadingIssues().insert(libPath);
100  }
101  return buildDefaultExtensionDefinition<T>();
102  }
103  getLibraries()[libPath] = extensionLibrary;
104  }
105  lock.unlock();
106 
107  std::string createName = "create" + std::string(name);
108  if (!extensionLibrary->has(createName)) {
109  // warning here because if the extension is not implemented, it may not be a problem for the simulation
110  Trace::warn() << DYNLog(IIDMExtensionNoCreate, name, libPath, createName) << Trace::endline;
111  return buildDefaultExtensionDefinition<T>();
112  }
113  auto createFunc = import<CreateFunctionBase<T>>(*extensionLibrary, createName);
114 
115  std::string destroyName = "destroy" + std::string(name);
116  if (!extensionLibrary->has(destroyName)) {
117  // warning here because if the extension is not implemented, it may not be a problem for the simulation
118  Trace::warn() << DYNLog(IIDMExtensionNoDestroy, name, libPath, destroyName) << Trace::endline;
119  return buildDefaultExtensionDefinition<T>();
120  }
121  auto destroyFunc = import<DestroyFunctionBase<T> >(*extensionLibrary, destroyName);
122 
123  return ExtensionDefinition<T>(createFunc, destroyFunc);
124  }
