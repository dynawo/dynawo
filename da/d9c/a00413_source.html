578  template<typename T>
579  std::shared_ptr<ParametersSet> addParameter(const std::string& name, T value, const std::string& row, const std::string& column) {
580  const std::vector<std::string>& parNames = tableParameterNames(name, row, column);
581  std::string firstParName;
582  bool isFirstParName = true;
583  for (std::vector<std::string>::const_iterator itName = parNames.begin(); itName != parNames.end(); ++itName) {
584  const std::string itParName = *itName;
585  if (isFirstParName) {
586  createParameter(itParName, value);
587  firstParName = itParName;
588  isFirstParName = false;
589  } else {
590  createAlias(itParName, firstParName);
591  }
592  }
593 
594  return shared_from_this();
595  }
