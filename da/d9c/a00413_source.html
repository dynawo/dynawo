579  template<typename T>
580  std::shared_ptr<ParametersSet> addParameter(const std::string& name, T value, const std::string& row, const std::string& column) {
581  const std::vector<std::string>& parNames = tableParameterNames(name, row, column);
582  std::string firstParName;
583  bool isFirstParName = true;
584  for (std::vector<std::string>::const_iterator itName = parNames.begin(); itName != parNames.end(); ++itName) {
585  const std::string itParName = *itName;
586  if (isFirstParName) {
587  createParameter(itParName, value);
588  firstParName = itParName;
589  isFirstParName = false;
590  } else {
591  createAlias(itParName, firstParName);
592  }
593  }
594 
595  return shared_from_this();
596  }
