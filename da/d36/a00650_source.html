29 template <typename T>
30 T ComponentInterface::getStaticParameterValue(const std::string& name) const {
31  boost::unordered_map<std::string, StaticParameter>::const_iterator iter = staticParameters_.find(name);
32  if (iter != staticParameters_.end()) {
33  StaticParameter::StaticParameterType type = iter->second.getType();
34  if (!iter->second.valueAffected()) {
35  throw DYNError(Error::MODELER, UnaffectedStaticParameter, name, getID());
36  } else {
37  switch (type) {
38  case StaticParameter::INT:
39  return static_cast<T>(iter->second.getValue<int>());
40  break;
41  case StaticParameter::DOUBLE:
42  return static_cast<T>(iter->second.getValue<double>());
43  break;
44  case StaticParameter::BOOL:
45  return static_cast<T>(iter->second.getValue<bool>());
46  break;
47  default:
48  throw DYNError(Error::MODELER, StaticParameterWrongType, name);
49  }
50  }
51  } else {
52  throw DYNError(Error::MODELER, UnknownStaticParameter, name, getID());
53  }
54 
55  return T();
56 }

58 template <typename T>
59 T ComponentInterface::getValue(const int index) const {
60 #ifdef _DEBUG_
61  if (checkStateVariableAreUpdatedBeforeCriteriaCheck_) {
62  assert(stateVariables_[index].isNeededForCriteriaCheck() && "Found a variable used in checkCriteria but not updated during model update.");
63  }
64 #endif
65  if (!stateVariables_[index].valueAffected()) {
66  throw DYNError(Error::MODELER, UnaffectedStateVariable, stateVariables_[index].getName(), getID());
67  } else {
68  switch (stateVariables_[index].getType()) {
69  case StateVariable::INT:
70  return static_cast<T>(stateVariables_[index].getValue<int>());
71  break;
72  case StateVariable::DOUBLE:
73  return static_cast<T>(stateVariables_[index].getValue<double>());
74  break;
75  case StateVariable::BOOL:
76  return static_cast<T>(stateVariables_[index].getValue<bool>());
77  break;
78  }
79  }
80  return T();
81 }
