    Test.cpp
)

          ${Boost_INCLUDE_DIR}
          ${LibIIDM_CORE_INCLUDE_DIR}

    iidm
    iidm-ext-standbyautomaton
    dynawo_DataInterface
    dynawo_ModelerCommon

  SET(EXTRACT_PATTERNS "'*/sources/Models/CPP/ModelNetwork/DYN*'")

  ADD_TEST_COVERAGE(${MODULE_NAME}-tests "${EXTRACT_PATTERNS}")
