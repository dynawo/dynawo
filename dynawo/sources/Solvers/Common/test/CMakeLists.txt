set(MODULE_NAME SOLVERCOMMON_unittest)

set(MODULE_SOURCES
    TestBasics.cpp
)

add_executable(${MODULE_NAME} ${MODULE_SOURCES})

target_link_libraries(${MODULE_NAME}
        dynawo_SolverCommon
        dynawo_ModelerCommon
        Sundials::Sundials_KINSOL
        dynawo_Test)

add_custom_target(${MODULE_NAME}-tests
  COMMAND ${CMAKE_COMMAND} -E env "${runtime_tests_PATH}" $<TARGET_FILE:${MODULE_NAME}>
  DEPENDS
    ${MODULE_NAME}
  COMMENT "Running ${MODULE_NAME}...")

if(CMAKE_BUILD_TYPE STREQUAL "TestCoverage")
  set(EXTRACT_PATTERNS "'*/sources/Solvers/Common/DYN*'")

  add_test_coverage(${MODULE_NAME}-tests "${EXTRACT_PATTERNS}")
endif()

if(CMAKE_BUILD_TYPE STREQUAL "Tests")
  add_test(${MODULE_NAME}-tests)
endif()
