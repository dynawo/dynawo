32 template <typename T>
33 T ComponentInterface::getStaticParameterValue(const std::string& name) const {
34  std::unordered_map<std::string, StaticParameter>::const_iterator iter = staticParameters_.find(name);
35  if (iter != staticParameters_.end()) {
36  StaticParameter::StaticParameterType type = iter->second.getType();
37  if (!iter->second.valueAffected()) {
38  throw DYNError(Error::MODELER, UnaffectedStaticParameter, name, getID());
39  } else {
40  switch (type) {
41  case StaticParameter::INT:
42  return static_cast<T>(iter->second.getValue<int>());
43  break;
44  case StaticParameter::DOUBLE:
45  return static_cast<T>(iter->second.getValue<double>());
46  break;
47  case StaticParameter::BOOL:
48  return static_cast<T>(iter->second.getValue<bool>());
49  break;
50  default:
51  throw DYNError(Error::MODELER, StaticParameterWrongType, name);
52  }
53  }
54  } else {
55  throw DYNError(Error::MODELER, UnknownStaticParameter, name, getID());
56  }
57 
58  return T();
59 }

61 template <typename T>
62 T ComponentInterface::getValue(const int index) const {
63 #ifdef _DEBUG_
64  if (checkStateVariableAreUpdatedBeforeCriteriaCheck_) {
65  assert(stateVariables_[index].isNeededForCriteriaCheck() && "Found a variable used in checkCriteria but not updated during model update.");
66  }
67 #endif
68  if (!stateVariables_[index].valueAffected()) {
69  throw DYNError(Error::MODELER, UnaffectedStateVariable, stateVariables_[index].getName(), getID());
70  } else {
71  switch (stateVariables_[index].getType()) {
72  case StateVariable::INT:
73  return static_cast<T>(stateVariables_[index].getValue<int>());
74  break;
75  case StateVariable::DOUBLE:
76  return static_cast<T>(stateVariables_[index].getValue<double>());
77  break;
78  case StateVariable::BOOL:
79  return static_cast<T>(stateVariables_[index].getValue<bool>());
80  break;
81  }
82  }
83  return T();
84 }
