169 inline modelica_real _event_div_real(modelica_real x1, modelica_real x2, modelica_integer index, DATA *data) {
170  modelica_real value1;
171  modelica_real value2;
172  if (data->simulationInfo->discreteCall && !data->simulationInfo->solveContinuous) {
173  data->simulationInfo->mathEventsValuePre[index] = x1;
174  data->simulationInfo->mathEventsValuePre[index+1] = x2;
175  }
176 
177  value1 = data->simulationInfo->mathEventsValuePre[index];
178  value2 = data->simulationInfo->mathEventsValuePre[index+1];
179 
180  return trunc(value1/value2);
181 }
