168 inline modelica_real _event_div_real(modelica_real x1, modelica_real x2, modelica_integer index, DATA *data) {
169  modelica_real value1;
170  modelica_real value2;
171  if (data->simulationInfo->discreteCall && !data->simulationInfo->solveContinuous) {
172  data->simulationInfo->mathEventsValuePre[index] = x1;
173  data->simulationInfo->mathEventsValuePre[index+1] = x2;
174  }
175 
176  value1 = data->simulationInfo->mathEventsValuePre[index];
177  value2 = data->simulationInfo->mathEventsValuePre[index+1];
178 
179  return trunc(value1/value2);
180 }
