59 inline NetworkComponent::findParameterDynamicNoThrow(const std::unordered_map<std::string, ParameterModeler>& params,
60  const std::string& id, bool& foundParam, const std::vector<std::string>& ids, T& value) const {
61  foundParam = false;
62  if (ids.empty()) {
63  if (hasParameter(id, params)) {
64  ParameterModeler pm = findParameter(id, params);
65  if (pm.hasOrigin(PAR)) {
66  value = pm.getValue<T>();
67  foundParam = true;
68  return;
69  }
70  }
71  } else {
72  for (std::vector<std::string>::const_iterator iter = ids.begin(), iterEnd = ids.end(); iter != iterEnd; ++iter) {
73  const std::string& name = (*iter) + "_" + id;
74  if (hasParameter(name, params)) {
75  ParameterModeler pm = findParameter(name, params);
76  if (pm.hasOrigin(PAR)) {
77  value = pm.getValue<T>();
78  foundParam = true;
79  return;
80  }
81  }
82  }
