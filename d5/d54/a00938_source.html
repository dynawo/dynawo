415  inline double ir_dUr(const double& ur, const double& ui, const double& U2,
416  const double& p, const double& q, const double& PdUr, const double& QdUr) const {
417  return ((PdUr * ur + p) + (QdUr * ui) - 2. * ur * (p * ur + q * ui) / U2) / U2;
418  }
419 
431  inline double ii_dUr(const double& ur, const double& ui, const double& U2,
432  const double& p, const double& q, const double& PdUr, const double& QdUr) const {
433  return ((PdUr * ui) - (QdUr * ur + q) - 2. * ur * (p * ui - q * ur) / U2) / U2;
434  }
435 
447  inline double ir_dUi(const double& ur, const double& ui, const double& U2,
448  const double& p, const double& q, const double& PdUi, const double& QdUi) const {
449  return ((PdUi * ur) + (QdUi * ui + q) - 2. * ui * (p * ur + q * ui) / U2) / U2;
450  }

463  inline double ii_dUi(const double& ur, const double & ui, const double& U2,
