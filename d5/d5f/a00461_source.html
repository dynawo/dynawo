137  static inline bool doubleEquals(const double a, const double b) {
138  if (std::isinf(a) || std::isinf(b)) return false;
139  if (std::isnan(a) || std::isnan(b)) return false;
140  double diff = std::fabs(a-b);
141  if (diff <= getCurrentPrecision()/5) return true;
142  return diff <= std::max(std::fabs(a), std::fabs(b)) * getCurrentPrecision() / 5.; // Error factor used to add more leeway
143  }
