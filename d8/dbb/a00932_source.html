390  inline double ir_dUr(const double& ur, const double& ui, const double& U2,
391  const double& p, const double& q, const double& PdUr, const double& QdUr) const {
392  return ((PdUr * ur + p) + (QdUr * ui) - 2. * ur * (p * ur + q * ui) / U2) / U2;

406  inline double ii_dUr(const double& ur, const double& ui, const double& U2,
407  const double& p, const double& q, const double& PdUr, const double& QdUr) const {
408  return ((PdUr * ui) - (QdUr * ur + q) - 2. * ur * (p * ui - q * ur) / U2) / U2;
409  }
