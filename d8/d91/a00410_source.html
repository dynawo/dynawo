572  template<typename T>
573  boost::shared_ptr<ParametersSet> addParameter(const std::string& name, T value, const std::string& row, const std::string& column) {
574  const std::vector<std::string>& parNames = tableParameterNames(name, row, column);
575  std::string firstParName;
576  bool isFirstParName = true;
577  for (std::vector<std::string>::const_iterator itName = parNames.begin(); itName != parNames.end(); ++itName) {
578  const std::string itParName = *itName;
579  if (isFirstParName) {
580  createParameter(itParName, value);
581  firstParName = itParName;
582  isFirstParName = false;
583  } else {
584  createAlias(itParName, firstParName);
585  }
586  }
587 
588  return shared_from_this();
589  }
