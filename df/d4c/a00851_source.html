460 inline void create_index_spec(index_spec_t* dest, int nridx, ...) {
461  int i;
462  va_list ap;
463  va_start(ap, nridx);
464 
465  dest->ndims = nridx;
466  dest->dim_size = size_alloc(nridx);
467  dest->index = index_alloc(nridx);
468  dest->index_type = reinterpret_cast<char*>(generic_alloc(nridx+1, sizeof(char)));
469  for (i = 0; i < nridx; ++i) {
470  dest->dim_size[i] = va_arg(ap, _index_t);
471  dest->index[i] = va_arg(ap, _index_t*);
472  dest->index_type[i] = static_cast<char>(va_arg(ap, _index_t)); /* char is cast to int by va_arg.*/
473  }
474  va_end(ap);
475 }
